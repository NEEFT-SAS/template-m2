src/
  main.ts                                   # Bootstrap NestJS (pipes, cors, swagger, global filters...)
  app.module.ts                              # Root module: assemble Core + tous les Contexts

  core/                                      # Modules techniques globaux reutilisables
    config/
      config.module.ts                       # Centralise ConfigModule Nest
      env.validation.ts                      # Validation des variables d'environnement

    database/
      database.module.ts                     # Connection DB (TypeORM) + providers
      typeorm.config.ts                      # Config TypeORM (entities, migrations, pool...)

    security/
      security.module.ts                     # Export JwtAuthGuard + JwtStrategy pour tous les contexts
      jwt/
        jwt.strategy.ts                      # Decode/valide JWT et hydrate req.user
        jwt-auth.guard.ts                    # Guard reutilisable (@UseGuards dans nimporte quel context)
        jwt.payload.ts                       # Typage payload JWT (sub, roles, scopes...)
      decorators/
        current-user.decorator.ts            # @CurrentUser() -> req.user
        public.decorator.ts                  # @Public() -> bypass guard (optionnel)

    mailer/
      mailer.module.ts                       # Module global envoi email
      mailer.service.ts                      # sendEmail(...) generique
      mailer.provider.ts                     # Adapter concret (nodemailer/sendgrid/mailjet...)

    http/
      exceptions/
        http-exception.filter.ts             # Formatage global des erreurs HTTP
      interceptors/
        response.interceptor.ts              # Standardise les responses (optionnel)

    logger/
      logger.module.ts                       # Module logger global
      logger.service.ts                      # Service logger (pino/winston)

    cache/
      cache.module.ts                        # Redis/cache (optionnel)
      cache.service.ts                       # get/set/invalidate

  contracts/                                 # Contrats partages (types/constants) entre contexts
    identity.contract.ts                     # Types partages pour user/roles (evite imports croises)
    recruitment.contract.ts                  # Types partages recruitment (status, enums...)

  contexts/                                  # 1 context = 1 mini-produit (feature-first)
    auth/
      auth.module.ts                         # Module AuthContext (providers + controllers)
      api/
        auth.controller.ts                   # Routes /auth (register/login/refresh/logout/reset...)
        auth.dtos.ts                         # DTO HTTP auth (RegisterDto/LoginDto/RefreshDto...)
      app/
        usecases/
          register.usecase.ts                # Creation compte + email verification (si active)
          login.usecase.ts                   # Login + creation session + tokens
          refresh.usecase.ts                 # Refresh token -> nouveau access token (rotation)
          logout.usecase.ts                  # Revoke session/refresh token
          request-password-reset.usecase.ts  # Genere token reset + envoi email
          reset-password.usecase.ts          # Reset password via token + update hash
        ports/
          auth.repository.port.ts            # Port unique DB auth (Account + Sessions)
          password-hasher.port.ts            # Port hash/compare password
          token.port.ts                      # Port generation/verification tokens
          mailer.port.ts                     # Port mail (bridge vers core/mailer)
      domain/
        entities/
          account.entity.ts                  # Aggregate Account/User (status, verified, etc.)
          session.entity.ts                  # Session (refresh token hash, revoke, expiresAt)
        value-objects/
          email.vo.ts                        # Normalisation + validation email
          password.vo.ts                     # Politique password (force, longueur...)
        errors/
          auth.errors.ts                     # InvalidCredentials, AccountLocked, EmailNotVerified...
      infra/
        persistence/
          auth.repository.typeorm.ts         # Impl TypeORM de AuthRepositoryPort
          orm/
            account.orm-entity.ts            # Entity TypeORM account
            session.orm-entity.ts            # Entity TypeORM session
        security/
          bcrypt-password-hasher.service.ts  # Impl PasswordHasherPort (bcrypt/argon2)
          token.service.ts                   # Impl TokenPort (JWT)

    players/
      players.module.ts                      # Module PlayersContext (profil joueur + features)
      api/
        players.controller.ts                # Routes /players (public + me) avec JwtAuthGuard si besoin
        players.dtos.ts                      # DTO player (update, add game, availability, media...)
      app/
        usecases/
          get-public-player-profile.usecase.ts  # Profil public (read model)
          get-my-player-profile.usecase.ts      # Profil prive du user connecte
          update-player-profile.usecase.ts      # Modif bio, socials, infos generales
          upsert-player-games.usecase.ts        # Gestion jeux/modes/ranks/saisons/roles/positions
          add-player-experience.usecase.ts      # Ajout experiences (hors membership officiel)
          update-player-availability.usecase.ts # Disponibilite recrutement
          add-player-media.usecase.ts           # Gallery media joueur
          manage-player-recommendations.usecase.ts # Recommendations (si interne)
        ports/
          players.repository.port.ts         # Repo unique players (profile+games+experiences+media)
          media.storage.port.ts              # Port stockage media (S3/local) optionnel
        queries/
          players.query.ts                   # Queries read-only optimise (profil public, listing...)
      domain/
        entities/
          player-profile.entity.ts           # Aggregate PlayerProfile
          player-game.entity.ts              # Sous-entite jeux (modes/ranks/saisons...)
          player-experience.entity.ts        # Experiences
        value-objects/
          player-slug.vo.ts                  # Normalisation slug player
          social-links.vo.ts                 # Social links (twitch, discord, twitter...)
        errors/
          players.errors.ts                  # PlayerNotFound, ForbiddenUpdate...
      infra/
        persistence/
          players.repository.typeorm.ts      # Impl TypeORM de PlayersRepositoryPort
          orm/
            player-profile.orm-entity.ts     # Entity TypeORM profil joueur
            player-game.orm-entity.ts        # Entity TypeORM jeux joueur
            player-experience.orm-entity.ts  # Entity TypeORM experiences
            player-media.orm-entity.ts       # Entity TypeORM gallery

    teams/
      teams.module.ts                        # Module TeamsContext (profil equipe + features)
      api/
        teams.controller.ts                  # Routes /teams (public + me + settings) avec JwtAuthGuard
        teams.dtos.ts                        # DTO team (branding, legal, socials, openings...)
      app/
        usecases/
          create-team.usecase.ts             # Creation team
          get-public-team-profile.usecase.ts # Profil public equipe
          get-my-team-profile.usecase.ts     # Profil prive equipe
          update-team-profile.usecase.ts     # Modif description, socials, etc.
          update-team-branding.usecase.ts    # Logo, banner, theme
          update-team-legal.usecase.ts       # RNA/SIREN + validation (si tu le fais ici)
          add-team-media.usecase.ts          # Gallery media team
          set-team-openings.usecase.ts       # Postes ouverts (annonces)
        ports/
          teams.repository.port.ts           # Repo unique teams (profile+legal+branding+media+openings)
        queries/
          teams.query.ts                     # Queries read-only optimise (profil public, listing...)
      domain/
        entities/
          team.entity.ts                     # Aggregate Team
          team-opening.entity.ts             # Postes ouverts / annonces
        value-objects/
          team-slug.vo.ts                    # Normalisation slug team
          legal-info.vo.ts                   # RNA/SIREN + structuration
        errors/
          teams.errors.ts                    # TeamNotFound, ForbiddenUpdate...
      infra/
        persistence/
          teams.repository.typeorm.ts        # Impl TypeORM de TeamsRepositoryPort
          orm/
            team.orm-entity.ts               # Entity TypeORM team
            team-opening.orm-entity.ts       # Entity TypeORM openings
            team-media.orm-entity.ts         # Entity TypeORM gallery
            team-legal.orm-entity.ts         # Entity TypeORM legal info (si table separee)

    recruitment/
      recruitment.module.ts                  # Context lien Player <-> Team (le plus important)
      api/
        recruitment.controller.ts            # Routes /recruitment (invite/apply/accept/reject/leave...)
        recruitment.dtos.ts                  # DTO pour candidatures et invitations
      app/
        usecases/
          invite-player.usecase.ts           # Team invite un player
          apply-to-team.usecase.ts           # Player candidate a une team
          accept-invite.usecase.ts           # Player accepte invitation
          reject-invite.usecase.ts           # Player refuse invitation
          accept-application.usecase.ts      # Team accepte candidature
          reject-application.usecase.ts      # Team refuse candidature
          leave-team.usecase.ts              # Player quitte une team
          kick-member.usecase.ts             # Team retire un membre (staff action)
        ports/
          recruitment.repository.port.ts     # Repo unique recruitment (membership + invites + applies)
          players.read.port.ts               # Port lecture player (public infos minimales)
          teams.read.port.ts                 # Port lecture team (public infos minimales)
        queries/
          recruitment.query.ts               # Queries roster/historique (optimise)
      domain/
        entities/
          team-membership.entity.ts          # Lien officiel Team <-> Player (status/role/dates)
          invitation.entity.ts               # Invitation envoyee par team
          application.entity.ts              # Candidature envoyee par player
        errors/
          recruitment.errors.ts              # AlreadyMember, InvitationNotFound, ForbiddenAction...
      infra/
        persistence/
          recruitment.repository.typeorm.ts  # Impl TypeORM RecruitmentRepositoryPort
          orm/
            team-membership.orm-entity.ts    # Entity TypeORM membership
            invitation.orm-entity.ts         # Entity TypeORM invitation
            application.orm-entity.ts        # Entity TypeORM application
        adapters/
          players.read.adapter.ts            # Adapter lecture PlayersContext (sans dépendance circulaire)
          teams.read.adapter.ts              # Adapter lecture TeamsContext (sans dépendance circulaire)

    messaging/
      messaging.module.ts                    # Context messagerie (si tu veux le sortir du reste)
      api/
        messaging.controller.ts              # Routes /messages (send, list, read...)
      app/
        usecases/
          send-message.usecase.ts            # Envoi message (gratuit/payant selon regles)
          list-conversations.usecase.ts      # Liste conversations
          read-conversation.usecase.ts       # Lecture messages
      domain/
        entities/
          conversation.entity.ts             # Conversation (1 par contact)
          message.entity.ts                  # Message
        errors/
          messaging.errors.ts                # PaymentRequired, CannotSendUntilReply...
      infra/
        persistence/
          messaging.repository.typeorm.ts    # Repo unique messaging
          orm/
            conversation.orm-entity.ts       # Entity TypeORM conversation
            message.orm-entity.ts            # Entity TypeORM message

    search/
      search.module.ts                       # Context recherche (query heavy)
      api/
        search.controller.ts                 # Routes /search (players/teams)
      app/
        queries/
          search-players.query.ts            # Recherche players (criteres avancés)
          search-teams.query.ts              # Recherche teams
      infra/
        persistence/
          search.read-repository.ts          # Repo read-only optimise (SQL + indexes + cache)
